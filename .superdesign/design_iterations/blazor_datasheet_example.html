@using BlazorDatasheet.Core.Data

@code {

}

💻 实际实现代码

在OrderDetail.razor中的完整实现：

1. Razor 标记部分：

@using BlazorDatasheet.Core.Data
@using BlazorDatasheet.Core.Data.Cells
@using BlazorDatasheet.DataStructures.Geometry
@using BlazorDatasheet.DataStructures.Store
@using BlazorDatasheet

<div style="margin-bottom: 20px;">
    <h4>Excel风格数据表格 (Ctrl+V 粘贴数据)</h4>
    <div class="datasheet-container">
        <Datasheet Sheet="@pasteDataSheet" />
    </div>
    <div style="margin-top: 8px; color: #666; font-size: 12px;">
        <Icon Type="@IconType.Outline.InfoCircle" style="margin-right: 4px;" />
        Excel风格表格：A列-货号，B列-数量，C列-单价，D列-折扣率，E列-总价。支持Ctrl+V粘贴Excel数据。
    </div>
</div>

2. C# 代码部分：

@code {
    // BlazorDatasheet 相关
    private Sheet pasteDataSheet = new(20, 5); // 20行5列的表格 (A, B, C, D, E)

    protected override async Task OnInitializedAsync()
    {
        InitializePasteDataSheet();
    }

    private void InitializePasteDataSheet()
    {
        // 创建新的表格，使用更多列模拟Excel风格
        pasteDataSheet = new Sheet(20, 5); // 20行5列：A, B, C, D, E列
        
        // 设置列标题为Excel风格的A, B, C, D, E
        pasteDataSheet.Cells[0, 0].Value = "ItemNo (A)";
        pasteDataSheet.Cells[0, 1].Value = "Quantity (B)"; 
        pasteDataSheet.Cells[0, 2].Value = "Price (C)";
        pasteDataSheet.Cells[0, 3].Value = "Discount (D)";
        pasteDataSheet.Cells[0, 4].Value = "Total (E)";
    }

    private void AddSampleData()
    {
        InitializePasteDataSheet();
        
        // 添加示例数据
        pasteDataSheet.Cells[1, 0].Value = "SAMPLE001";
        pasteDataSheet.Cells[1, 1].Value = "10";
        pasteDataSheet.Cells[1, 2].Value = "25.50";
        pasteDataSheet.Cells[1, 3].Value = "0.1";
        pasteDataSheet.Cells[1, 4].Value = "22.95";
        
        pasteDataSheet.Cells[2, 0].Value = "SAMPLE002";
        pasteDataSheet.Cells[2, 1].Value = "5";
        pasteDataSheet.Cells[2, 2].Value = "15.00";
        pasteDataSheet.Cells[2, 3].Value = "0";
        pasteDataSheet.Cells[2, 4].Value = "75.00";
    }
}
